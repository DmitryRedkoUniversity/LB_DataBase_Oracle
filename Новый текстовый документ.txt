CREATE TABLE SCHOOL
(
    SCHOOLNAME CHARACTER VARYING(30) PRIMARY KEY,
    DISTRICT CHARACTER VARYING(30) REFERENCES district (schooldistrict)
);

CREATE TABLE SCHOOLBUILDINGS
(
    BUILDINGNAME CHARACTER VARYING(30),
	SCHOOLNAME CHARACTER VARYING(30) REFERENCES school (SCHOOLNAME),
    PRIMARY KEY(BUILDINGNAME, SCHOOLNAME)
);

CREATE TABLE DISTRICT
(
    SCHOOLDISTRICT CHARACTER VARYING(30) PRIMARY KEY,
    DISTRICTPHONENUMBER CHARACTER VARYING(30),
	FIOSCHOOLCURATOR CHARACTER VARYING(30)
);
